main.rs 
use fantasy_craft::prelude::*;

// Define a component
#[derive(Component)]
struct Player {
    speed: f32,
}

#[fantasy_main]
async fn main() {
    let mut world = World::new();
    
    world.spawn((
        Player { speed: 10.0 },
        Transform::from_xyz(0.0, 5.0, 0.0),
        Sprite::load("hero.png").await,
    ));

    // Start the game loop
    Engine::run(world).await;
}
